36 int KineticSocket_Connect(const char* host, int port, bool blocking)
37 {
38  char message[256];
39  char port_str[32];
40  struct addrinfo* result = NULL;
41  struct addrinfo hints;
42  struct addrinfo* ai = NULL;
43  int socket_fd;
44  int res;
45 
46  sprintf(message, "Connecting to %s:%d", host, port);
47  LOG(message);
48 
49  memset(&hints, 0, sizeof(struct addrinfo));
50 
51  // could be inet or inet6
52  hints.ai_family = PF_UNSPEC;
53  hints.ai_socktype = SOCK_STREAM;
54  hints.ai_protocol = IPPROTO_TCP;
55  hints.ai_flags = AI_NUMERICSERV;
56 
57  sprintf(port_str, "%d", port);
58 
59  res = getaddrinfo(host, port_str, &hints, &result);
60  if (res != 0)
61  {
62  LOG("Could not resolve host");
63  return -1;
64  }
65 
66  for (ai = result; ai != NULL; ai = ai->ai_next)
67  {
68  char host[NI_MAXHOST];
69  char service[NI_MAXSERV];
70  int res;
71 
72  res = getnameinfo(ai->ai_addr, ai->ai_addrlen, host, sizeof(host), service,
73  sizeof(service), NI_NUMERICHOST | NI_NUMERICSERV);
74  if (res != 0)
75  {
76  LOG("Could not get name info");
77  continue;
78  }
79  else
80  {
81  LOG("Trying to connect to host");
82  }
83 
84  socket_fd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol);
85  if (socket_fd == -1)
86  {
87  LOG("Could not create socket");
88  continue;
89  }
90 
91  // os x won't let us set close-on-exec when creating the socket, so set it separately
92  int current_fd_flags = fcntl(socket_fd, F_GETFD);
93  if (current_fd_flags == -1)
94  {
95  LOG("Failed to get socket fd flags");
96  close(socket_fd);
97  continue;
98  }
99  if (fcntl(socket_fd, F_SETFD, current_fd_flags | FD_CLOEXEC) == -1)
100  {
101  LOG("Failed to set socket close-on-exit");
102  close(socket_fd);
103  continue;
104  }
105 
106  // On BSD-like systems we can set SO_NOSIGPIPE on the socket to prevent it from sending a
107  // PIPE signal and bringing down the whole application if the server closes the socket
108  // forcibly
109 // #ifdef SO_NOSIGPIPE
110 // int set = 1;
111 // int setsockopt_result = setsockopt(socket_fd, SOL_SOCKET, SO_NOSIGPIPE, &set, sizeof(set));
112 // // Allow ENOTSOCK because it allows tests to use pipes instead of real sockets
113 // if (setsockopt_result != 0 && setsockopt_result != ENOTSOCK)
114 // {
115 // LOG("Failed to set SO_NOSIGPIPE on socket");
116 // close(socket_fd);
117 // continue;
118 // }
119 // #endif
120 
121  if (connect(socket_fd, ai->ai_addr, ai->ai_addrlen) == -1)
122  {
123  LOG("Unable to connect");
124  close(socket_fd);
125  continue;
126  }
127  else
128  {
129  LOG("Connected to host!");
130  }
131 
132  if (!blocking && fcntl(socket_fd, F_SETFL, O_NONBLOCK) != 0)
133  {
134  LOG("Failed to set socket nonblocking");
135  close(socket_fd);
136  continue;
137  }
138 
139  break;
140  }
141 
142  freeaddrinfo(result);
143 
144  if (ai == NULL)
145  {
146  // we went through all addresses without finding one we could bind to
147  sprintf(message, "Could not connect to %s:%d", host, port);
148  LOG(message);
149  return -1;
150  }
151 
152  return socket_fd;
153 }

173 bool KineticSocket_Read(int socketDescriptor, void* buffer, size_t length)
174 {
175  size_t count;
176  uint8_t* data = (uint8_t*)buffer;
177 
178  for (count = 0; count < length; )
179  {
180  int status;
181  fd_set readSet;
182  struct timeval timeout;
183 
184  // Time out after 5 seconds
185  timeout.tv_sec = 5;
186  timeout.tv_usec = 0;
187 
188  FD_ZERO(&readSet);
189  FD_SET(socketDescriptor, &readSet);
190  status = select(socketDescriptor+1, &readSet, NULL, NULL, &timeout);
191 
192  if (status < 0) // Error occurred
193  {
194  char msg[128];
195  sprintf(msg, "Failed waiting to read from socket! status=%d, errno=%d\n", status, errno);
196  LOG(msg);
197  return false;
198  }
199  else if (status == 0) // Timeout occurred
200  {
201  LOG("Timed out waiting for socket data to arrive!");
202  return false;
203  }
204  else if (status > 0) // Data available to read
205  {
206  // The socket is ready for reading
207  status = read(socketDescriptor, &buffer[count], length - count);
208  if (status == -1 && errno == EINTR)
209  {
210  continue;
211  }
212  else if (status <= 0)
213  {
214  char msg[128];
215  sprintf(msg, "Failed to read from socket! status=%d, errno=%d\n", status, errno);
216  LOG(msg);
217  return false;
218  }
219  else
220  {
221  count += status;
222  }
223  }
224  }
225 
226  return true;
227 }

229 bool KineticSocket_ReadProtobuf(int socketDescriptor, KineticProto* message, void* buffer, size_t length)
230 {
231  bool success = false;
232  KineticProto* proto;
233 
234  if (KineticSocket_Read(socketDescriptor, buffer, length))
235  {
236  proto = KineticProto_unpack(NULL, length, buffer);
237  if (proto == NULL)
238  {
239  LOG("Error unpacking incoming Kinetic protobuf message!");
240  }
241  else
242  {
243  assert(message != NULL);
244  memcpy(message, proto, length);
245  KineticProto_free_unpacked(proto, NULL);
246  success = true;
247  }
248  }

253 bool KineticSocket_Write(int socketDescriptor, const void* buffer, size_t length)
254 {
255  int status;
256  size_t count;
257  uint8_t* data = (uint8_t*)buffer;
258 
259  for (count = 0; count < length; count += status)
260  {
261  status = write(socketDescriptor, &buffer[count], length - count);
262  if (status == -1 && errno == EINTR)
263  {
264  continue;
265  }
266  else if (status <= 0)
267  {
268  char msg[128];
269  sprintf(msg, "Failed to write to socket! status=%d, errno=%d\n", status, errno);
270  LOG(msg);
271  return false;
272  }
273  }
274 
275  return true;
276 }
