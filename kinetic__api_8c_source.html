105 KineticProto_Status_StatusCode KineticApi_NoOp(KineticOperation* operation)
106 {
107  KineticProto_Status_StatusCode status =
108  KINETIC_PROTO_STATUS_STATUS_CODE_INVALID_STATUS_CODE;
109 
110  assert(operation->exchange != NULL);
111  assert(operation->exchange->connection != NULL);
112  assert(operation->request != NULL);
113  assert(operation->request->protobuf != NULL);
114  assert(operation->response != NULL);
115  assert(operation->response->protobuf != NULL);
116 
117  // Initialize request
118  KineticExchange_IncrementSequence(operation->exchange);
119  KineticMessage_Init(operation->request->protobuf);
120  KineticPDU_Init(operation->request, operation->request->exchange, operation->request->protobuf, NULL, 0);
121  KineticOperation_BuildNoop(operation);
122 
123  // Send the request
124  KineticPDU_Send(operation->request);
125 
126  // Associate response with same exchange as request
127  KineticPDU_Init(operation->response, operation->response->exchange, operation->response->protobuf, NULL, 0);
128 
129  // Receive the response
130  if (KineticPDU_Receive(operation->response))
131  {
132  status = operation->response->protobuf->command.status->code;
133  }
134 
135  return status;
136 }
